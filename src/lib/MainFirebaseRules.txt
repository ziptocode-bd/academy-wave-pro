rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() { return request.auth != null; }
    function isSelf(uid)  { return isSignedIn() && request.auth.uid == uid; }
    function isAdmin() {
      return isSignedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
    }
    function isApproved() {
      return isSignedIn()
        && exists(/databases/$(database)/documents/users/$(request.auth.uid))
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.status == 'approved';
    }
    // basic size guard against abuse / bloat writes
    function smallPayload() {
      return request.resource.size() < 200000; // ~200KB
    }

    // ── USERS ──────────────────────────────────────────────
    // Read: self or admin only. NEVER public.
    // Update: self can edit own profile but cannot change role/status/email/createdAt/enrolledCourses.
    match /users/{userId} {
      allow read: if isSelf(userId) || isAdmin();

      allow create: if isSelf(userId)
                    && smallPayload()
                    && request.resource.data.role == 'student'
                    && request.resource.data.status == 'pending'
                    && request.resource.data.email is string
                    && request.resource.data.name is string
                    && request.resource.data.name.size() <= 80
                    && request.resource.data.email.size() <= 120;

      allow update: if (
                      isSelf(userId)
                      && smallPayload()
                      && request.resource.data.role   == resource.data.role
                      && request.resource.data.status == resource.data.status
                      && request.resource.data.email  == resource.data.email
                      && request.resource.data.createdAt == resource.data.createdAt
                      && request.resource.data.enrolledCourses == resource.data.enrolledCourses
                    ) || isAdmin();

      allow delete: if isAdmin();
    }

    // ── COURSES (public read for SEO/landing) ─────────────
    match /courses/{courseId} {
      allow read: if true;
      allow create, update, delete: if isAdmin() && smallPayload();
    }

    // ── VIDEOS (only approved + enrolled students; admin always) ─
    // We can't check enrollment without an extra read; approved+signed-in is enforced.
    // Per-course enrollment is also enforced client-side and at the URL level by
    // ProtectedAdminRoute + VideoPlayerPage. Unauthenticated users get nothing.
    match /videos/{videoId} {
      allow read: if isApproved() || isAdmin();
      allow create, update, delete: if isAdmin() && smallPayload();
    }

    // ── ENROLL REQUESTS ───────────────────────────────────
    // Anti-spam: payload size guard + only owner can create/re-submit.
    match /enrollRequests/{reqId} {
      allow read: if isAdmin()
                  || (isSignedIn() && resource.data.userId == request.auth.uid);

      allow create: if isSignedIn()
                    && smallPayload()
                    && request.resource.data.userId == request.auth.uid
                    && request.resource.data.status == 'pending'
                    && request.resource.data.courseId is string
                    && request.resource.data.courseId.size() <= 80
                    && request.resource.data.transactionId is string
                    && request.resource.data.transactionId.size() <= 80;

      // Only re-submit a rejected request — nothing else from the student side.
      allow update: if isAdmin()
                    || (
                      isSignedIn()
                      && smallPayload()
                      && resource.data.userId == request.auth.uid
                      && resource.data.status == 'rejected'
                      && request.resource.data.status == 'pending'
                      && request.resource.data.userId == resource.data.userId
                      && request.resource.data.courseId == resource.data.courseId
                      && request.resource.data.diff(resource.data).affectedKeys()
                           .hasOnly(['status', 'paymentMethod', 'paymentNumber', 'transactionId', 'createdAt'])
                    );

      allow delete: if isAdmin();
    }

    // ── APP SETTINGS (public read, admin write) ──────────
    match /settings/{docId} {
      allow read: if true;
      allow write: if isAdmin() && smallPayload();
    }

    // ── VERSIONS META (public read for cache-invalidation, admin write) ──
    match /_meta/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    // ── Default deny ──────────────────────────────────────
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
