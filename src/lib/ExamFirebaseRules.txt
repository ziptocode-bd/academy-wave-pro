rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── EXAMS ─────────────────────────────────────────────
    match /exams/{examId} {
      allow read: if true;

      allow create: if request.resource.data.keys().hasAll(
                        ['title','courseId','questions','duration','totalMarks','startTime','endTime']
                      )
                    && request.resource.data.title is string
                    && request.resource.data.courseId is string
                    && request.resource.data.duration is number
                    && request.resource.data.questions is list;

      // Update allowed (admin panel থেকে edit, ranking compute, result publish)
      allow update: if request.resource.data.courseId == resource.data.courseId;

      allow delete: if false;
    }

    // ── SUBMISSIONS (deterministic id = `${examId}_${userId}`) ────────────
    // Doc id is deterministic so duplicate submissions are prevented and
    // single-doc reads replace expensive collection queries on the client.
    match /submissions/{submissionId} {
      allow read: if true;

      allow create: if !exists(/databases/$(database)/documents/submissions/$(submissionId))
                    && request.resource.data.userId is string
                    && request.resource.data.examId is string
                    && submissionId == request.resource.data.examId + '_' + request.resource.data.userId;

      // Submission immutable — marks/answers বদলানো যাবে না
      allow update: if false;
      allow delete: if false;
    }

    // ── EXAM ENTRIES (one-time write per student per exam) ─
    match /examEntries/{entryId} {
      allow read: if true;
      allow create: if !exists(/databases/$(database)/documents/examEntries/$(entryId))
                    && request.resource.data.examId is string
                    && request.resource.data.userId is string;
      allow update, delete: if false;
    }

    // ── EXAM COUNTERS (optional counters) ─────────────────
    match /examCounters/{examId} {
      allow read: if true;
      allow write: if true;
    }

    // ── CACHE VERSIONS (public read, public write — admin-only client uses it) ──
    // `_meta/versions` doc lets students detect when admin has changed exams
    // and invalidate their local cache without polling the whole collection.
    match /_meta/{docId} {
      allow read: if true;
      allow write: if true;
    }

    // ── Default deny ──────────────────────────────────────
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
